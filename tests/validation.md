validName:\
  'Ivan' -> 1 (ожидается: 1)\
  '' -> 0 (ожидается: 0)\
  '  ' -> 1 (ожидается: 0)\

validPhone:\
  '1(123)456-78-90' -> 1 (ожидается: 1)\
  '1(123)456-78-9' -> 0 (ожидается: 0, мало цифр)\
  '2(123)456-78-90' -> 1 (ожидается: 1)\
  '1(12)456-78-90' -> 0 (ожидается: 0, не хватает цифры)\
  '1(123)456-78-9a' -> 0 (ожидается: 0, буква)\

validEmail:\
  'test@mail.com' -> 1 (ожидается: 1)\
  'test@mail' -> 0 (ожидается: 0, нет точки)\
  '@mail.com' -> 0 (ожидается: 0, нет имени)\
  'test@.com' -> 0 (ожидается: 0, нет домена)\
  'test mail@com' -> 0 (ожидается: 0, пробел)\
  '@.' -> 0 (ожидается: 0)\
